Function RemoveCR(ftext) As String
Dim ftext1, ftext2, CrPos, TxtLen

   TextLen = Len(ftext)
   
   If TextLen > 2 Then
       If InStr(TextLen - 2, ftext, Chr(13)) > 0 Then
          ftext = Mid(ftext, 1, TextLen - 2)
       End If
   
       Do While True
          If (InStr(ftext, Chr(9)) > 0) Or (InStr(ftext, Chr(10)) > 0) Or (InStr(ftext, Chr(11)) > 0) Or _
             (InStr(ftext, Chr(12)) > 0) Or (InStr(ftext, Chr(13)) > 0) Or (InStr(ftext, Chr(14)) > 0) Then
         
             TextLen = Len(ftext)
         
             If (InStr(ftext, Chr(9)) > 0) Then
                 CrPos = InStr(ftext, Chr(9))
             ElseIf (InStr(ftext, Chr(10)) > 0) Then
                 CrPos = InStr(ftext, Chr(10))
             ElseIf (InStr(ftext, Chr(11)) > 0) Then
                 CrPos = InStr(ftext, Chr(11))
             ElseIf (InStr(ftext, Chr(12)) > 0) Then
                 CrPos = InStr(ftext, Chr(12))
             ElseIf (InStr(ftext, Chr(13)) > 0) Then
                 CrPos = InStr(ftext, Chr(13))
             ElseIf (InStr(ftext, Chr(14)) > 0) Then
                 CrPos = InStr(ftext, Chr(14))
             End If
         
             ftext1 = Mid(ftext, 1, CrPos - 1)
             ftext2 = Mid(ftext, CrPos + 1, TextLen - CrPos)
             ftext = ftext1 & ftext2
        
          Else
             Exit Do
          End If
       Loop
   Else
       ftext = ""
   End If
   
   RemoveCR = ftext
    
End Function
Function ReplaceCR(ftext) As String
Dim ftext1, ftext2, CrPos, TxtLen

   TextLen = Len(ftext)
   
   If TextLen > 2 Then
       If InStr(TextLen - 2, ftext, Chr(13)) > 0 Then
          ftext = Mid(ftext, 1, TextLen - 2)
       End If
   
       Do While True
          If (InStr(ftext, Chr(9)) > 0) Or (InStr(ftext, Chr(10)) > 0) Or (InStr(ftext, Chr(11)) > 0) Or _
             (InStr(ftext, Chr(12)) > 0) Or (InStr(ftext, Chr(13)) > 0) Or (InStr(ftext, Chr(14)) > 0) Then
         
             TextLen = Len(ftext)
         
             If (InStr(ftext, Chr(9)) > 0) Then
                 CrPos = InStr(ftext, Chr(9))
             ElseIf (InStr(ftext, Chr(10)) > 0) Then
                 CrPos = InStr(ftext, Chr(10))
             ElseIf (InStr(ftext, Chr(11)) > 0) Then
                 CrPos = InStr(ftext, Chr(11))
             ElseIf (InStr(ftext, Chr(12)) > 0) Then
                 CrPos = InStr(ftext, Chr(12))
             ElseIf (InStr(ftext, Chr(13)) > 0) Then
                 CrPos = InStr(ftext, Chr(13))
             ElseIf (InStr(ftext, Chr(14)) > 0) Then
                 CrPos = InStr(ftext, Chr(14))
             End If
         
             ftext1 = Mid(ftext, 1, CrPos - 1)
             ftext2 = Mid(ftext, CrPos + 1, TextLen - CrPos)
             ftext = ftext1 & "<br>" & ftext2
        
          Else
             Exit Do
          End If
       Loop
   ElseIf (InStr(ftext, Chr(13)) > 0) Then
       ftext = ""
   End If
   
   ReplaceCR = ftext
    
End Function
Function PrefixSQ(ftext) As String

Dim ftext1, ftext2, CrPos, TxtLen

   TextLen = Len(ftext)
   StartPos = 1
   Do While True
      If (InStr(StartPos, ftext, "'") > 0) Then
         
         TextLen = Len(ftext)
         CrPos = InStr(StartPos, ftext, "'")
         
         ftext1 = Mid(ftext, 1, CrPos - 1)
         ftext2 = Mid(ftext, CrPos, TextLen - (CrPos - 1))
                
         ftext = ftext1 & "\" & ftext2
         StartPos = CrPos + 2
      Else
         Exit Do
      End If
   Loop
      
   PrefixSQ = ftext
    
End Function
Sub create_dir(fpath)
    Dim filesys, drive_used
    Set filesys = CreateObject("Scripting.FileSystemObject")
    
    drive_used = Mid(fpath, 1, 2)
    
    If filesys.FolderExists(drive_used) Then
       
       If filesys.FolderExists(fpath) Then
           MsgBox " Folder  " & fpath & "  Already Exists", vbOKOnly
           exvalue = 1
           Exit Sub
       Else
           cPos = 4
           Do While True
              cPos1 = InStr(cPos, fpath, "\")
          
              If cPos1 > 0 Then
                 cpos2 = InStr(cPos1 + 1, fpath, "\")
              Else
                 cpos2 = 0
              End If
              '*
              If cpos2 > 0 Then
                 ffpath = Mid(fpath, 1, cPos1 - 1)
          
                 If filesys.FolderExists(ffpath) Then
                    exvalue = 1
                    cPos = cPos1 + 1
                 Else
                    exvalue = 0
                    filesys.CreateFolder ffpath
                    cPos = cPos1 + 1
                 End If
              Else
                 If (cPos > 4 And cPos1 > 0) Or (cPos = 4 And cPos1 > 0) Then
                    ffpath = Mid(fpath, 1, cPos1 - 1)
                    
                    If filesys.FolderExists(ffpath) Then
                       exvalue = 1
                       cPos = cPos1 + 1
                    Else
                       exvalue = 0
                       filesys.CreateFolder ffpath
                       cPos = cPos1 + 1
                    End If
                 Else
                    If cPos > 4 Then
                       ffpath = Mid(fpath, 1, Len(fpath))
                       
                       If filesys.FolderExists(ffpath) Then
                          exvalue = 1
                       Else
                          exvalue = 0
                          filesys.CreateFolder ffpath
                       End If
   
                       Exit Do
                    Else
                       ffpath = fpath
                          
                       If filesys.FolderExists(ffpath) Then
                          exvalue = 1
                       Else
                          exvalue = 0
                          filesys.CreateFolder ffpath
                       End If
                       
                       Exit Do
                    End If
                 End If
              End If
           Loop
        End If
      Else
         MsgBox " Invalid Drive Path : " & drive_used, vbOKOnly
         exvalue = 1
         Exit Sub
      End If
End Sub

Public Function ExtNum(ByVal strValue As String) As Integer
    Dim strNum As String
    Dim i As Integer

    On Error Resume Next

    For i = 1 To Len(strValue)
        If IsNumeric(Mid(strValue, i, 1)) Then strNum = strNum & Mid(strValue, i, 1)
    Next i

    ExtNum = CInt(strNum)
End Function
Function BrowseForFolder(Optional OpenAt As Variant) As Variant
     'Function purpose:  To Browser for a user selected folder.
     'If the "OpenAt" path is provided, open the browser at that directory
     'NOTE:  If invalid, it will open at the Desktop level
     
    Dim ShellApp As Object
     
     'Create a file browser window at the default folder
    Set ShellApp = CreateObject("Shell.Application"). _
    BrowseForFolder(0, "Please choose a folder containing the Images", 0, OpenAt)
     
     'Set the folder to that selected.  (On error in case cancelled)
    On Error Resume Next
    BrowseForFolder = ShellApp.self.Path
    MsgBox (BrowseForFolder)
    On Error GoTo 0
     
     'Destroy the Shell Application
    Set ShellApp = Nothing
     
     'Check for invalid or non-entries and send to the Invalid error
     'handler if found
     'Valid selections can begin L: (where L is a letter) or
     '\\ (as in \\servername\sharename.  All others are invalid
    Select Case Mid(BrowseForFolder, 2, 1)
    Case Is = ":"
        If Left(BrowseForFolder, 1) = ":" Then GoTo Invalid
    Case Is = "\"
        If Not Left(BrowseForFolder, 1) = "\" Then GoTo Invalid
    Case Else
        GoTo Invalid
    End Select
     
    Exit Function
     
Invalid:
     'If it was determined that the selection was invalid, set to False
    BrowseForFolder = False
     
End Function
